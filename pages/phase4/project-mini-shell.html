Project: Mini Shell

Project Goal

Build a minimal Unix shell that can execute programs, handle built-in commands (cd, exit), and chain commands with pipes. This project teaches how Bash and zsh work internally — fork() creates a child process, exec() replaces it with a new program, and wait() lets the parent collect the exit status. Every terminal session you've ever opened runs this exact pattern.

⚠ Platform Note

This project uses POSIX system calls (fork, execvp, wait, pipe) available on Linux and macOS. It does not work on Windows without WSL2. This is intentional — learning POSIX C is part of systems programming.

Project Specification

  • Print a prompt, read a command line, execute it
  • Built-in commands: cd <dir>, pwd, exit, help
  • Execute external programs using fork() + execvp()
  • Handle Ctrl+C gracefully (don't kill the shell)
  • Support pipes: ls | grep .c | wc -l
  • Return exit status of last command (echo $? equivalent)

Key Concepts: fork, exec, wait

How a shell runs "ls -la": Shell process (PID 1234) │ ├─ fork() ─────────────────────────────────┐ │ │ │ Parent process (PID 1234) │ Child process (PID 1235) │ - wait(NULL) ← blocks until child done │ - execvp("ls", ["ls","-la",NULL]) │ │ - process image replaced with ls │ │ - ls runs, prints output │ │ - ls exits with code 0 │ │ │ ←──────────────────────────────────────┤ │ wait() returns, child is reaped │ │ last_exit_status = 0 │ │ print prompt again │ fork() return values: - In parent process: returns child's PID (> 0) - In child process: returns 0 - On error: returns -1

Step 1: Read-Eval-Print Loop

/* shell.c — Step 1: basic read-eval loop
   Compile: gcc -Wall -o shell shell.c */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>
#include <errno.h>

#define MAX_ARGS 64
#define MAX_LINE 1024

/* Split line into argv[] array, returns argc */
int parse_args(char *line, char *argv[], int max_args) {
    int argc = 0;
    char *tok = strtok(line, " \t\n");
    while (tok && argc < max_args - 1) {
        argv[argc++] = tok;
        tok = strtok(NULL, " \t\n");
    }
    argv[argc] = NULL;
    return argc;
}

int main(void) {
    char line[MAX_LINE];
    char *argv[MAX_ARGS];

    while (1) {
        printf("mysh$ ");
        fflush(stdout);

        if (!fgets(line, sizeof(line), stdin)) {
            printf("\n");
            break;  /* EOF (Ctrl+D) */
        }

        int argc = parse_args(line, argv, MAX_ARGS);
        if (argc == 0) continue;

        /* TODO: handle built-ins and execute commands */
        printf("Command: %s, %d args\n", argv[0], argc);
    }

    return 0;
}

Step 2: Built-in Commands

/* Returns 1 if command was a built-in (handled), 0 if not */
int handle_builtin(char *argv[], int *last_status) {
    if (!argv[0]) return 0;

    if (strcmp(argv[0], "exit") == 0) {
        int code = argv[1] ? atoi(argv[1]) : 0;
        exit(code);
    }

    if (strcmp(argv[0], "cd") == 0) {
        const char *dir = argv[1] ? argv[1] : getenv("HOME");
        if (!dir) dir = "/";
        if (chdir(dir) != 0) {
            fprintf(stderr, "cd: %s: %s\n", dir, strerror(errno));
            *last_status = 1;
        } else {
            *last_status = 0;
        }
        return 1;
    }

    if (strcmp(argv[0], "pwd") == 0) {
        char cwd[1024];
        if (getcwd(cwd, sizeof(cwd))) {
            printf("%s\n", cwd);
            *last_status = 0;
        } else {
            perror("pwd");
            *last_status = 1;
        }
        return 1;
    }

    if (strcmp(argv[0], "help") == 0) {
        printf("Built-ins: cd, pwd, exit, help\n");
        printf("Run any program: ls, grep, gcc, etc.\n");
        *last_status = 0;
        return 1;
    }

    return 0;  /* not a built-in */
}

Step 3: fork + execvp

#include <unistd.h>
#include <sys/wait.h>

int execute(char *argv[]) {
    pid_t pid = fork();

    if (pid < 0) {
        perror("fork");
        return -1;
    }

    if (pid == 0) {
        /* --- CHILD PROCESS --- */
        execvp(argv[0], argv);
        /* execvp only returns on failure */
        fprintf(stderr, "%s: %s\n", argv[0], strerror(errno));
        exit(127);  /* 127 = command not found convention */
    }

    /* --- PARENT PROCESS --- */
    int status;
    waitpid(pid, &status, 0);

    if (WIFEXITED(status)) {
        return WEXITSTATUS(status);   /* normal exit */
    }
    if (WIFSIGNALED(status)) {
        fprintf(stderr, "Killed by signal %d\n", WTERMSIG(status));
        return 128 + WTERMSIG(status);
    }
    return -1;
}

Step 4: Pipes

/* Execute a two-command pipeline: cmd1 | cmd2
   e.g. parse "ls | grep .c" into two argv arrays */

int execute_pipe(char *argv1[], char *argv2[]) {
    int pipefd[2];  /* pipefd[0] = read end, pipefd[1] = write end */
    if (pipe(pipefd) < 0) { perror("pipe"); return -1; }

    pid_t pid1 = fork();
    if (pid1 == 0) {
        /* Child 1: runs cmd1, writes to pipe */
        dup2(pipefd[1], STDOUT_FILENO);  /* redirect stdout to pipe write end */
        close(pipefd[0]);
        close(pipefd[1]);
        execvp(argv1[0], argv1);
        perror(argv1[0]);
        exit(127);
    }

    pid_t pid2 = fork();
    if (pid2 == 0) {
        /* Child 2: runs cmd2, reads from pipe */
        dup2(pipefd[0], STDIN_FILENO);   /* redirect stdin to pipe read end */
        close(pipefd[0]);
        close(pipefd[1]);
        execvp(argv2[0], argv2);
        perror(argv2[0]);
        exit(127);
    }

    /* Parent: close both ends, wait for both children */
    close(pipefd[0]);
    close(pipefd[1]);

    int status1, status2;
    waitpid(pid1, &status1, 0);
    waitpid(pid2, &status2, 0);

    return WIFEXITED(status2) ? WEXITSTATUS(status2) : -1;
}

/* Split command line at '|' */
int has_pipe(char *line) {
    return strchr(line, '|') != NULL;
}

/* Parse "ls -la | grep .c" into two argv arrays */
void parse_pipe(char *line, char *argv1[], char *argv2[]) {
    char *pipe_pos = strchr(line, '|');
    *pipe_pos = '\0';   /* split the string at | */
    parse_args(line, argv1, MAX_ARGS);
    parse_args(pipe_pos + 1, argv2, MAX_ARGS);
}

Complete Mini Shell

/* shell.c — Complete mini shell
   Compile: gcc -Wall -o mysh shell.c
   Run: ./mysh */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>
#include <errno.h>
#include <signal.h>

#define MAX_ARGS 64
#define MAX_LINE 1024

/* Ignore SIGINT in the shell (so Ctrl+C doesn't kill it) */
void setup_signals(void) {
    signal(SIGINT, SIG_IGN);   /* shell ignores Ctrl+C */
    signal(SIGQUIT, SIG_IGN);  /* shell ignores Ctrl+\ */
}

int parse_args(char *line, char *argv[], int max) {
    int n = 0;
    char *t = strtok(line, " \t\n");
    while (t && n < max - 1) { argv[n++] = t; t = strtok(NULL, " \t\n"); }
    argv[n] = NULL;
    return n;
}

int handle_builtin(char *argv[], int *status) {
    if (!argv[0]) return 0;
    if (!strcmp(argv[0], "exit"))  { exit(argv[1] ? atoi(argv[1]) : 0); }
    if (!strcmp(argv[0], "cd")) {
        const char *d = argv[1] ?: getenv("HOME") ?: "/";
        *status = chdir(d) ? (perror("cd"), 1) : 0;
        return 1;
    }
    if (!strcmp(argv[0], "pwd")) {
        char cwd[1024];
        if (getcwd(cwd, sizeof(cwd))) { puts(cwd); *status = 0; }
        else { perror("pwd"); *status = 1; }
        return 1;
    }
    return 0;
}

int execute_cmd(char *argv[]) {
    pid_t pid = fork();
    if (pid < 0) { perror("fork"); return -1; }
    if (pid == 0) {
        signal(SIGINT, SIG_DFL);  /* child restores default Ctrl+C */
        execvp(argv[0], argv);
        fprintf(stderr, "%s: %s\n", argv[0], strerror(errno));
        exit(127);
    }
    int st; waitpid(pid, &st, 0);
    return WIFEXITED(st) ? WEXITSTATUS(st) : -1;
}

int main(void) {
    setup_signals();
    char line[MAX_LINE];
    char *argv[MAX_ARGS], *argv2[MAX_ARGS];
    int last_status = 0;

    while (1) {
        char cwd[256]; getcwd(cwd, sizeof(cwd));
        printf("\033[32mmysh\033[0m:\033[34m%s\033[0m$ ", cwd);
        fflush(stdout);

        if (!fgets(line, sizeof(line), stdin)) { puts(""); break; }
        line[strcspn(line, "\n")] = '\0';
        if (!*line) continue;

        char line_copy[MAX_LINE];
        strncpy(line_copy, line, sizeof(line_copy) - 1);

        if (strchr(line_copy, '|')) {
            char *pipe = strchr(line_copy, '|');
            *pipe = '\0';
            int pfd[2]; pipe(pfd);
            /* run left side writing to pipe, right side reading from pipe */
            parse_args(line_copy, argv, MAX_ARGS);
            parse_args(pipe + 1, argv2, MAX_ARGS);
            /* simplified: execute_pipe as shown above */
            last_status = execute_pipe(argv, argv2);
        } else {
            parse_args(line_copy, argv, MAX_ARGS);
            if (!handle_builtin(argv, &last_status)) {
                last_status = execute_cmd(argv);
            }
        }
    }
    return 0;
}
