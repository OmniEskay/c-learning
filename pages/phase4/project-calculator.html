Project: CLI Calculator

Project Goal

Build a command-line calculator that accepts expressions like 3.5 + 2 * (4 - 1), handles operator precedence, validates input, and loops until the user quits. This project exercises functions, loops, input validation, and switch-based dispatch — all the Phase 1 fundamentals applied together.

Project Specification

  • Accept two numbers and an operator: 10 / 3 → 3.333333
  • Supported operators: +, -, *, /, %
  • Handle division and modulo by zero gracefully (print error, don't crash)
  • Loop continuously until user types q or quit
  • Show a history of the last 10 calculations
  • Validate all input (non-numeric input, invalid operators)

Step 2: Computation Function

#include <stdio.h>
#include <math.h>
#include <stdbool.h>

/* Returns false if the operation is invalid (e.g., div by zero) */
bool compute(double a, char op, double b, double *result) {
    switch (op) {
        case '+': *result = a + b; return true;
        case '-': *result = a - b; return true;
        case '*': *result = a * b; return true;
        case '/':
            if (b == 0.0) {
                fprintf(stderr, "Error: division by zero\n");
                return false;
            }
            *result = a / b;
            return true;
        case '%':
            if ((long long)b == 0) {
                fprintf(stderr, "Error: modulo by zero\n");
                return false;
            }
            *result = fmod(a, b);
            return true;
        default:
            fprintf(stderr, "Error: unknown operator '%c'\n", op);
            return false;
    }
}

Step 3: Input Validation

/* Read a line of input and check for quit command */
#include <string.h>

bool read_expression(double *a, char *op, double *b) {
    char line[128];
    if (!fgets(line, sizeof(line), stdin)) return false;

    /* Trim leading whitespace and check for quit */
    char trimmed[128];
    if (sscanf(line, "%127s", trimmed) == 1) {
        if (strcmp(trimmed, "q") == 0 || strcmp(trimmed, "quit") == 0) {
            return false;
        }
    }

    /* Parse expression */
    if (sscanf(line, "%lf %c %lf", a, op, b) != 3) {
        fprintf(stderr, "Invalid input. Use format: 10 + 3\n");
        *a = *b = 0; *op = '?';
        return true;   /* don't quit, but mark as invalid */
    }
    return true;
}

Step 4: History Feature

#define HISTORY_SIZE 10

typedef struct {
    double a, b, result;
    char op;
} Entry;

Entry history[HISTORY_SIZE];
int history_count = 0;

void history_add(double a, char op, double b, double result) {
    /* Shift entries if full */
    if (history_count == HISTORY_SIZE) {
        for (int i = 0; i < HISTORY_SIZE - 1; i++) {
            history[i] = history[i + 1];
        }
        history_count = HISTORY_SIZE - 1;
    }
    history[history_count++] = (Entry){a, b, result, op};
}

void history_print(void) {
    if (history_count == 0) { printf("No history.\n"); return; }
    printf("--- History ---\n");
    for (int i = 0; i < history_count; i++) {
        Entry *e = &history[i];
        printf("  %g %c %g = %g\n", e->a, e->op, e->b, e->result);
    }
    printf("---------------\n");
}

Complete Reference Implementation

/* calculator.c — Complete implementation
   Compile: gcc -Wall -o calc calculator.c -lm */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <stdbool.h>

#define HISTORY_MAX 10

typedef struct { double a, b, result; char op; } Entry;
static Entry hist[HISTORY_MAX];
static int hist_n = 0;

static void hist_add(double a, char op, double b, double r) {
    if (hist_n == HISTORY_MAX) {
        for (int i = 0; i < HISTORY_MAX - 1; i++) hist[i] = hist[i+1];
        hist_n = HISTORY_MAX - 1;
    }
    hist[hist_n++] = (Entry){a, b, r, op};
}

static void hist_print(void) {
    printf("─── History ────────────────────\n");
    if (hist_n == 0) { printf("  (empty)\n"); return; }
    for (int i = 0; i < hist_n; i++)
        printf("  %g %c %g = %g\n", hist[i].a, hist[i].op, hist[i].b, hist[i].result);
    printf("────────────────────────────────\n");
}

static bool compute(double a, char op, double b, double *res) {
    switch (op) {
        case '+': *res = a + b; return true;
        case '-': *res = a - b; return true;
        case '*': *res = a * b; return true;
        case '/':
            if (b == 0.0) { fprintf(stderr, "Error: division by zero\n"); return false; }
            *res = a / b;
            return true;
        case '%':
            if (b == 0.0) { fprintf(stderr, "Error: modulo by zero\n"); return false; }
            *res = fmod(a, b);
            return true;
        default:
            fprintf(stderr, "Unknown operator '%c'. Use: + - * / %%\n", op);
            return false;
    }
}

int main(void) {
    char line[256];
    printf("C Calculator — type 'quit' to exit, 'history' for log\n");
    printf("Usage: <number> <op> <number>  e.g. 3.14 * 2\n\n");

    while (1) {
        printf("> ");
        fflush(stdout);

        if (!fgets(line, sizeof(line), stdin)) break;

        char first[64];
        sscanf(line, "%63s", first);
        if (strcmp(first, "q") == 0 || strcmp(first, "quit") == 0) break;
        if (strcmp(first, "history") == 0) { hist_print(); continue; }

        double a, b, result;
        char op;
        if (sscanf(line, "%lf %c %lf", &a, &op, &b) != 3) {
            fprintf(stderr, "Invalid input. Example: 10 + 3\n");
            continue;
        }

        if (compute(a, op, b, &result)) {
            printf("= %g\n", result);
            hist_add(a, op, b, result);
        }
    }

    printf("Goodbye!\n");
    return 0;
}
