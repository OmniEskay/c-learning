Project: File-Backed Records System

Project Goal

Extend the Student Records system to persist data to disk. Records are stored in a binary file, loaded at startup, and saved on exit. Add search (by name or GPA range) and sort capabilities. This is how embedded databases work at their simplest level.

Project Specification

  • All features of Project 2 (Student Records)
  • Auto-save to students.dat binary file on every change
  • Load from file at startup (create if not found)
  • Search by name substring or GPA range: search gpa 3.0 4.0
  • Export to CSV: export students.csv
  • Import from CSV file

Step 1: File Format and Load/Save

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <errno.h>

#define DB_FILE "students.dat"
#define MAGIC   0x43534442  /* "CSDB" */
#define VERSION 1

/* File header — written at the start of the binary file */
typedef struct {
    unsigned int magic;    /* sanity check */
    int          version;
    int          count;
    int          next_id;
} FileHeader;

int db_save(const Records *r) {
    FILE *fp = fopen(DB_FILE, "wb");
    if (!fp) { perror("save"); return -1; }

    FileHeader hdr = {MAGIC, VERSION, r->size, r->next_id};
    if (fwrite(&hdr, sizeof(hdr), 1, fp) != 1) {
        perror("write header"); fclose(fp); return -1;
    }
    if (r->size > 0) {
        if ((int)fwrite(r->data, sizeof(Student), r->size, fp) != r->size) {
            perror("write records"); fclose(fp); return -1;
        }
    }
    if (fclose(fp) != 0) { perror("fclose"); return -1; }
    return 0;
}

int db_load(Records *r) {
    FILE *fp = fopen(DB_FILE, "rb");
    if (!fp) {
        if (errno == ENOENT) return 0;  /* file doesn't exist — OK */
        perror("load"); return -1;
    }

    FileHeader hdr;
    if (fread(&hdr, sizeof(hdr), 1, fp) != 1) {
        fprintf(stderr, "Corrupt database (bad header)\n");
        fclose(fp); return -1;
    }
    if (hdr.magic != MAGIC) {
        fprintf(stderr, "Not a valid database file\n");
        fclose(fp); return -1;
    }

    /* Allocate space for loaded records */
    if (hdr.count > 0) {
        Student *data = malloc(hdr.count * sizeof(Student));
        if (!data) { fclose(fp); return -1; }
        if ((int)fread(data, sizeof(Student), hdr.count, fp) != hdr.count) {
            fprintf(stderr, "Corrupt database (truncated)\n");
            free(data); fclose(fp); return -1;
        }
        free(r->data);
        r->data = data;
        r->size = r->capacity = hdr.count;
        r->next_id = hdr.next_id;
    }
    fclose(fp);
    printf("Loaded %d record(s) from %s\n", hdr.count, DB_FILE);
    return 0;
}

Step 2: CSV Export and Import

int csv_export(const Records *r, const char *path) {
    FILE *fp = fopen(path, "w");
    if (!fp) { perror("export"); return -1; }

    fprintf(fp, "id,name,age,gpa\n");   /* header row */
    for (int i = 0; i < r->size; i++) {
        const Student *s = &r->data[i];
        fprintf(fp, "%d,\"%s\",%d,%.2f\n", s->id, s->name, s->age, s->gpa);
    }

    if (fclose(fp) != 0) { perror("close"); return -1; }
    printf("Exported %d records to %s\n", r->size, path);
    return 0;
}

int csv_import(Records *r, const char *path) {
    FILE *fp = fopen(path, "r");
    if (!fp) { perror("import"); return -1; }

    char line[256];
    fgets(line, sizeof(line), fp);  /* skip header */

    int imported = 0;
    while (fgets(line, sizeof(line), fp)) {
        int id_unused, age;
        double gpa;
        char name[NAME_MAX];
        /* Parse: id,"name",age,gpa */
        if (sscanf(line, "%*d,\"%49[^\"]\",%d,%lf", name, &age, &gpa) == 3) {
            records_add(r, name, age, gpa);
            imported++;
        }
    }
    fclose(fp);
    printf("Imported %d records from %s\n", imported, path);
    return 0;
}

Step 3: Advanced Search

void records_search_gpa(const Records *r, double lo, double hi) {
    int found = 0;
    printf("Students with GPA in [%.2f, %.2f]:\n", lo, hi);
    print_header();
    for (int i = 0; i < r->size; i++) {
        if (r->data[i].gpa >= lo && r->data[i].gpa <= hi) {
            print_record(&r->data[i]);
            found++;
        }
    }
    printf("└─────────────────────────────────────────────────┘\n");
    printf("%d result(s)\n", found);
}

/* Parse the user's search command */
void handle_search(Records *r, const char *cmd) {
    /* "search name Alice" or "search gpa 3.0 4.0" */
    char type[16], arg1[64], arg2[32];
    int parts = sscanf(cmd, "search %15s %63s %31s", type, arg1, arg2);

    if (parts < 2) { printf("Usage: search name <query> | search gpa <lo> <hi>\n"); return; }

    if (strcmp(type, "name") == 0) {
        records_search(r, arg1);
    } else if (strcmp(type, "gpa") == 0 && parts == 3) {
        records_search_gpa(r, atof(arg1), atof(arg2));
    } else {
        printf("Unknown search type '%s'\n", type);
    }
}

Complete main with Command Parser

/* main loop for file-backed records
   Compile: gcc -Wall -o filedb main.c */

int main(void) {
    Records *r = records_new();
    if (!r) return 1;
    db_load(r);   /* load from disk at startup */

    char line[256];
    printf("File Records System. Commands: add, remove <id>, search, display,\n");
    printf("  stats, sort gpa|name, export <file>, import <file>, quit\n\n");

    while (1) {
        printf("db> "); fflush(stdout);
        if (!fgets(line, sizeof(line), stdin)) break;
        line[strcspn(line, "\n")] = '\0';  /* strip trailing newline */

        if (strcmp(line, "quit") == 0 || strcmp(line, "q") == 0) break;
        else if (strcmp(line, "display") == 0) records_display(r);
        else if (strcmp(line, "stats") == 0) records_stats(r);
        else if (strcmp(line, "sort gpa") == 0) { records_sort_by_gpa(r); db_save(r); }
        else if (strcmp(line, "sort name") == 0) { records_sort_by_name(r); db_save(r); }
        else if (strncmp(line, "search", 6) == 0) handle_search(r, line);
        else if (strncmp(line, "export ", 7) == 0) csv_export(r, line + 7);
        else if (strncmp(line, "import ", 7) == 0) { csv_import(r, line + 7); db_save(r); }
        else if (strcmp(line, "add") == 0) {
            char name[NAME_MAX]; int age; double gpa;
            printf("Name: "); fgets(name, NAME_MAX, stdin); name[strcspn(name,"\n")]='\0';
            printf("Age: "); fgets(line, sizeof(line), stdin); age = atoi(line);
            printf("GPA: "); fgets(line, sizeof(line), stdin); gpa = atof(line);
            int id = records_add(r, name, age, gpa);
            printf("Added ID %d\n", id);
            db_save(r);
        }
        else if (strncmp(line, "remove ", 7) == 0) {
            int id = atoi(line + 7);
            printf(records_remove(r, id) ? "Removed.\n" : "Not found.\n");
            db_save(r);
        }
        else printf("Unknown command. Type 'quit' to exit.\n");
    }

    db_save(r);
    records_free(r);
    return 0;
}
