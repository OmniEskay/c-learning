Project: Student Records System

Project Goal

Build an in-memory student records system using structs and a dynamic array. This project demonstrates how real databases work at their most basic level: records are structs, operations are functions that operate on arrays of structs, and queries are loops with conditionals.

Project Specification

  • Store student records: ID, name, age, GPA
  • Operations: add student, remove by ID, search by name, display all
  • Dynamic array that grows as needed (no fixed limit)
  • Print a formatted table with column headers
  • Compute and display class statistics: average GPA, highest/lowest
  • Sort records by GPA (descending) or name (alphabetical)

Step 1: Data Model

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define NAME_MAX 50

typedef struct {
    int    id;
    char   name[NAME_MAX];
    int    age;
    double gpa;
} Student;

typedef struct {
    Student *data;
    int      size;
    int      capacity;
    int      next_id;
} Records;

Records *records_new(void) {
    Records *r = malloc(sizeof(Records));
    if (!r) return NULL;
    r->data = malloc(4 * sizeof(Student));
    if (!r->data) { free(r); return NULL; }
    r->size = 0;
    r->capacity = 4;
    r->next_id = 1;
    return r;
}

void records_free(Records *r) {
    if (!r) return;
    free(r->data);
    free(r);
}

Step 2: Add and Remove

/* Returns the new student's ID, or -1 on failure */
int records_add(Records *r, const char *name, int age, double gpa) {
    /* Grow if needed */
    if (r->size == r->capacity) {
        int newcap = r->capacity * 2;
        Student *tmp = realloc(r->data, newcap * sizeof(Student));
        if (!tmp) return -1;
        r->data = tmp;
        r->capacity = newcap;
    }

    Student *s = &r->data[r->size];
    s->id  = r->next_id++;
    s->age = age;
    s->gpa = gpa;
    strncpy(s->name, name, NAME_MAX - 1);
    s->name[NAME_MAX - 1] = '\0';
    r->size++;
    return s->id;
}

/* Returns 1 if found and removed, 0 if not found */
int records_remove(Records *r, int id) {
    for (int i = 0; i < r->size; i++) {
        if (r->data[i].id == id) {
            /* Shift remaining records down */
            for (int j = i; j < r->size - 1; j++) {
                r->data[j] = r->data[j + 1];
            }
            r->size--;
            return 1;
        }
    }
    return 0;
}

Step 3: Search and Display

/* Search by name (case-insensitive partial match) */
void records_search(const Records *r, const char *query) {
    int found = 0;
    printf("Search results for '%s':\n", query);
    print_header();
    for (int i = 0; i < r->size; i++) {
        /* Use strcasestr if available, or manual search */
        if (strstr(r->data[i].name, query)) {
            print_record(&r->data[i]);
            found++;
        }
    }
    if (found == 0) printf("  (no results)\n");
    else printf("%d result(s)\n", found);
}

static void print_header(void) {
    printf("┌─────┬──────────────────────┬─────┬───────┐\n");
    printf("│ ID  │ Name                 │ Age │ GPA   │\n");
    printf("├─────┼──────────────────────┼─────┼───────┤\n");
}

static void print_record(const Student *s) {
    printf("│ %-3d │ %-20s │ %-3d │ %.2f  │\n",
           s->id, s->name, s->age, s->gpa);
}

void records_display(const Records *r) {
    if (r->size == 0) { printf("No records.\n"); return; }
    print_header();
    for (int i = 0; i < r->size; i++) print_record(&r->data[i]);
    printf("└─────┴──────────────────────┴─────┴───────┘\n");
    printf("%d student(s)\n", r->size);
}

Step 4: Sort and Statistics

#include <stdlib.h>

/* qsort comparator: descending GPA */
static int cmp_gpa_desc(const void *a, const void *b) {
    const Student *sa = a, *sb = b;
    if (sb->gpa > sa->gpa) return 1;
    if (sb->gpa < sa->gpa) return -1;
    return 0;
}

/* qsort comparator: alphabetical by name */
static int cmp_name(const void *a, const void *b) {
    return strcmp(((const Student *)a)->name, ((const Student *)b)->name);
}

void records_sort_by_gpa(Records *r) {
    qsort(r->data, r->size, sizeof(Student), cmp_gpa_desc);
}

void records_sort_by_name(Records *r) {
    qsort(r->data, r->size, sizeof(Student), cmp_name);
}

void records_stats(const Records *r) {
    if (r->size == 0) { printf("No data.\n"); return; }
    double sum = 0, hi = r->data[0].gpa, lo = r->data[0].gpa;
    int hi_idx = 0, lo_idx = 0;
    for (int i = 0; i < r->size; i++) {
        sum += r->data[i].gpa;
        if (r->data[i].gpa > hi) { hi = r->data[i].gpa; hi_idx = i; }
        if (r->data[i].gpa < lo) { lo = r->data[i].gpa; lo_idx = i; }
    }
    printf("Students: %d\n", r->size);
    printf("Avg GPA:  %.2f\n", sum / r->size);
    printf("Highest:  %s (%.2f)\n", r->data[hi_idx].name, hi);
    printf("Lowest:   %s (%.2f)\n", r->data[lo_idx].name, lo);
}

Complete Reference Implementation

/* main.c — Student Records main loop
   Compile: gcc -Wall -o records main.c */

int main(void) {
    Records *r = records_new();
    if (!r) { fprintf(stderr, "Out of memory\n"); return 1; }

    /* Seed some data */
    records_add(r, "Alice Chen",   20, 3.85);
    records_add(r, "Bob Martinez", 22, 3.20);
    records_add(r, "Carol Singh",  21, 3.95);
    records_add(r, "David Kim",    23, 2.75);

    char line[128];
    int running = 1;

    while (running) {
        printf("\n1. Display all  2. Add  3. Remove  4. Search\n");
        printf("5. Sort by GPA  6. Sort by name  7. Stats  0. Quit\n");
        printf("Choice: ");
        fflush(stdout);

        if (!fgets(line, sizeof(line), stdin)) break;
        int choice = atoi(line);

        switch (choice) {
            case 1: records_display(r); break;
            case 2: {
                char name[NAME_MAX]; int age; double gpa;
                printf("Name: "); fgets(name, NAME_MAX, stdin);
                name[strcspn(name, "\n")] = '\0';
                printf("Age: "); fgets(line, sizeof(line), stdin);
                age = atoi(line);
                printf("GPA: "); fgets(line, sizeof(line), stdin);
                gpa = atof(line);
                int id = records_add(r, name, age, gpa);
                printf("Added with ID %d\n", id);
                break;
            }
            case 3: {
                printf("Remove ID: "); fgets(line, sizeof(line), stdin);
                int id = atoi(line);
                printf(records_remove(r, id) ? "Removed.\n" : "Not found.\n");
                break;
            }
            case 4: {
                printf("Search name: "); fgets(line, sizeof(line), stdin);
                line[strcspn(line, "\n")] = '\0';
                records_search(r, line);
                break;
            }
            case 5: records_sort_by_gpa(r); records_display(r); break;
            case 6: records_sort_by_name(r); records_display(r); break;
            case 7: records_stats(r); break;
            case 0: running = 0; break;
            default: printf("Unknown option.\n");
        }
    }

    records_free(r);
    printf("Goodbye!\n");
    return 0;
}
