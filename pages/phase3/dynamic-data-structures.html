Dynamic Data Structures

Why does this matter?

The Linux kernel's task scheduler uses a red-black tree (a balanced BST) of runnable processes. The kernel's memory allocator is a slab allocator built on linked lists. Redis stores sorted sets using skip lists. Understanding these data structures — and implementing them from scratch in C — is what separates systems programmers from application programmers.

Singly Linked List

A linked list is a chain of nodes, each holding a value and a pointer to the next node. Unlike arrays, nodes need not be contiguous in memory — you can insert and delete without shifting elements.

Singly Linked List: 10 → 20 → 30 → NULL ┌────────┬──────────┐ ┌────────┬──────────┐ ┌────────┬──────────┐ │ val: 10│ next ────┼──▶│ val: 20│ next ────┼──▶│ val: 30│ next: NULL│ └────────┴──────────┘ └────────┴──────────┘ └────────┴──────────┘ Node at 0x600010 Node at 0x600030 Node at 0x600050 head = 0x600010 To traverse: start at head, follow ->next until NULL To insert at head: new_node->next = head; head = new_node; To delete a node: prev->next = node->next; free(node);
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int value;
    struct Node *next;
} Node;

/* Create a new node */
Node *node_new(int val) {
    Node *n = malloc(sizeof(Node));
    if (!n) return NULL;
    n->value = val;
    n->next = NULL;
    return n;
}

/* Insert at the head — O(1) */
Node *list_prepend(Node *head, int val) {
    Node *n = node_new(val);
    if (!n) return head;
    n->next = head;
    return n;
}

/* Append to the tail — O(n) */
Node *list_append(Node *head, int val) {
    Node *n = node_new(val);
    if (!n) return head;
    if (!head) return n;
    Node *cur = head;
    while (cur->next) cur = cur->next;
    cur->next = n;
    return head;
}

/* Print all values */
void list_print(Node *head) {
    for (Node *cur = head; cur; cur = cur->next) {
        printf("%d", cur->value);
        if (cur->next) printf(" → ");
    }
    printf(" → NULL\n");
}

/* Delete first node with given value — O(n) */
Node *list_delete(Node *head, int val) {
    if (!head) return NULL;
    if (head->value == val) {
        Node *next = head->next;
        free(head);
        return next;
    }
    Node *cur = head;
    while (cur->next && cur->next->value != val) {
        cur = cur->next;
    }
    if (cur->next) {
        Node *del = cur->next;
        cur->next = del->next;
        free(del);
    }
    return head;
}

/* Free the entire list */
void list_free(Node *head) {
    while (head) {
        Node *next = head->next;
        free(head);
        head = next;
    }
}

int main(void) {
    Node *list = NULL;
    list = list_append(list, 10);
    list = list_append(list, 20);
    list = list_append(list, 30);
    list = list_prepend(list, 5);

    list_print(list);       /* 5 → 10 → 20 → 30 → NULL */
    list = list_delete(list, 20);
    list_print(list);       /* 5 → 10 → 30 → NULL */

    list_free(list);
    return 0;
}

Stack (LIFO)

Stack (Last In, First Out): push(1) → push(2) → push(3) top ┌───────────────────────────────┐ │ │ [3] ← most recent pushed │ ◀─┘ ├───────────────────────────────┤ │ [2] │ ├───────────────────────────────┤ │ [1] ← first pushed │ └───────────────────────────────┘ pop() → returns 3 (top) pop() → returns 2 pop() → returns 1 pop() → empty!
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

typedef struct StackNode {
    int value;
    struct StackNode *next;
} StackNode;

typedef struct { StackNode *top; int size; } Stack;

Stack stack_new(void) { return (Stack){NULL, 0}; }

bool stack_push(Stack *s, int val) {
    StackNode *n = malloc(sizeof(StackNode));
    if (!n) return false;
    n->value = val;
    n->next = s->top;
    s->top = n;
    s->size++;
    return true;
}

bool stack_pop(Stack *s, int *out) {
    if (!s->top) return false;
    StackNode *n = s->top;
    *out = n->value;
    s->top = n->next;
    free(n);
    s->size--;
    return true;
}

bool stack_peek(const Stack *s, int *out) {
    if (!s->top) return false;
    *out = s->top->value;
    return true;
}

void stack_free(Stack *s) {
    int tmp;
    while (stack_pop(s, &tmp));
}

int main(void) {
    Stack s = stack_new();
    stack_push(&s, 1);
    stack_push(&s, 2);
    stack_push(&s, 3);

    int val;
    while (stack_pop(&s, &val)) {
        printf("%d\n", val);  /* 3, 2, 1 */
    }
    return 0;
}

Queue (FIFO)

Queue (First In, First Out): enqueue(A) → enqueue(B) → enqueue(C) front rear │ │ ▼ ▼ ┌───┐ ┌───┐ ┌───┐ ┌───┐ ┌───┐ │ A │──▶│ B │──▶│ C │──▶│...│──▶│ │ └───┘ └───┘ └───┘ └───┘ └───┘ dequeue() → returns A (front) dequeue() → returns B
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

typedef struct QNode { int val; struct QNode *next; } QNode;
typedef struct { QNode *front; QNode *rear; int size; } Queue;

Queue queue_new(void) { return (Queue){NULL, NULL, 0}; }

bool queue_enqueue(Queue *q, int val) {
    QNode *n = malloc(sizeof(QNode));
    if (!n) return false;
    n->val = val;
    n->next = NULL;
    if (!q->rear) {
        q->front = q->rear = n;
    } else {
        q->rear->next = n;
        q->rear = n;
    }
    q->size++;
    return true;
}

bool queue_dequeue(Queue *q, int *out) {
    if (!q->front) return false;
    QNode *n = q->front;
    *out = n->val;
    q->front = n->next;
    if (!q->front) q->rear = NULL;
    free(n);
    q->size--;
    return true;
}

Doubly Linked List

Doubly Linked List: NULL ← 10 ↔ 20 ↔ 30 → NULL prev val next prev val next NULL ←─┬────┬──────┬────┬─────────┬────┬──────┬────┬─→ NULL │ │ 10 │ ──┼────────▶│ │ 20 │ ──┼──▶... │ ◀──┼──────┤ │ │ ◀──┼──────┤ │ └────┴──────┴────┘ └────┴──────┴────┘ Advantages over singly linked list: - O(1) deletion given a pointer to the node (no need to find prev) - Can traverse backwards
typedef struct DNode {
    int value;
    struct DNode *prev;
    struct DNode *next;
} DNode;

DNode *dlist_insert_after(DNode *node, int val) {
    DNode *n = malloc(sizeof(DNode));
    if (!n) return NULL;
    n->value = val;
    n->next = node->next;
    n->prev = node;
    if (node->next) node->next->prev = n;
    node->next = n;
    return n;
}

void dlist_remove(DNode *node) {
    if (node->prev) node->prev->next = node->next;
    if (node->next) node->next->prev = node->prev;
    free(node);
}
⚠ Common Mistakes
  • Not freeing the entire list — When you're done with a list, you must traverse and free every node. Freeing only the head leaks all other nodes.
  • Forgetting to handle the empty-list case — Many list operations (delete, print, append) must check head == NULL first.
  • NULL dereference on last node's next — The last node's next is NULL. Always check before accessing cur->next->value.

Practice Exercises

Basic Reverse a linked list

Write a function Node *list_reverse(Node *head) that reverses a singly linked list in-place (without allocating new nodes) and returns the new head.

Show hint
Use three pointers: prev = NULL, cur = head, next. Loop: save next = cur->next; set cur->next = prev; advance prev = cur; cur = next. Return prev when cur is NULL.
Intermediate Detect a cycle in a linked list

Implement Floyd's cycle detection algorithm (tortoise and hare). Write a function that returns true if a linked list contains a cycle, false otherwise.

Show hint
Use two pointers: slow advances 1 step at a time, fast advances 2 steps. If they meet, there's a cycle. If fast or fast->next becomes NULL, there's no cycle.
Challenge Implement a circular buffer

Implement a fixed-capacity circular buffer (ring buffer) using an array. It should support: push (fails when full), pop (fails when empty), peek, size, is_full, is_empty. No dynamic allocation — use a fixed-size array in the struct.

Show hint
Store head and tail indices. Push: buf[tail] = val; tail = (tail+1) % capacity; Pop: val = buf[head]; head = (head+1) % capacity. Full when (tail+1)%capacity == head. Empty when head == tail.

Checkpoint Quiz

Test your understanding

1. In a singly linked list, the last node's next pointer should be:
NULL signals the end of the list. A pointer back to head would create a circular list — useful sometimes, but a bug if unintentional.
2. A stack follows which order?
Stacks are Last-In First-Out. Think of a stack of plates — you take from the top (last placed). Queues are FIFO.
3. What is the time complexity of inserting at the head of a singly linked list?
Head insertion is O(1): new_node->next = head; head = new_node. No traversal needed. Tail insertion is O(n) without a tail pointer.
