File I/O

Why does this matter?

Every database, log system, and configuration parser is built on file I/O. SQLite's entire storage engine is a set of carefully crafted fread/fwrite calls. When you understand the C file I/O model, you understand why disk operations are the bottleneck in most systems — and why databases use techniques like write-ahead logging and memory-mapped files to work around it.

Opening and Closing Files

#include <stdio.h>

int main(void) {
    /* fopen returns a FILE* — NULL if it fails */
    FILE *fp = fopen("data.txt", "w");

    if (fp == NULL) {          /* ALWAYS check for NULL */
        perror("fopen failed");
        return 1;
    }

    fprintf(fp, "Hello, file!\n");
    fprintf(fp, "Line 2\n");

    fclose(fp);   /* ALWAYS close files when done */
    return 0;
}

Text File Reading

#include <stdio.h>

int main(void) {
    FILE *fp = fopen("data.txt", "r");
    if (!fp) { perror("open"); return 1; }

    /* Method 1: read character by character */
    int ch;
    while ((ch = fgetc(fp)) != EOF) {   /* fgetc returns int (not char!) */
        putchar(ch);
    }
    rewind(fp);   /* seek back to start */

    /* Method 2: read line by line */
    char line[256];
    while (fgets(line, sizeof(line), fp) != NULL) {
        printf("Line: %s", line);   /* fgets keeps the \n */
    }
    rewind(fp);

    /* Method 3: read formatted values */
    int n;
    float f;
    while (fscanf(fp, "%d %f", &n, &f) == 2) {   /* check return value! */
        printf("Read: %d %.2f\n", n, f);
    }

    fclose(fp);
    return 0;
}

Binary File I/O

Use fread/fwrite to read and write raw bytes — perfect for storing structs as binary records:

#include <stdio.h>

typedef struct { char name[30]; int score; } Record;

int main(void) {
    /* Write binary records */
    Record records[] = {
        {"Alice", 95},
        {"Bob", 82},
        {"Carol", 91},
    };
    int n = sizeof(records) / sizeof(records[0]);

    FILE *fp = fopen("scores.bin", "wb");
    if (!fp) { perror("write"); return 1; }
    fwrite(records, sizeof(Record), n, fp);   /* write n records */
    fclose(fp);

    /* Read them back */
    fp = fopen("scores.bin", "rb");
    if (!fp) { perror("read"); return 1; }

    Record r;
    while (fread(&r, sizeof(Record), 1, fp) == 1) {   /* read 1 record at a time */
        printf("%-20s %d\n", r.name, r.score);
    }

    fclose(fp);
    return 0;
}

File Position with fseek and ftell

#include <stdio.h>

int main(void) {
    FILE *fp = fopen("data.bin", "rb");
    if (!fp) return 1;

    /* Get file size */
    fseek(fp, 0, SEEK_END);          /* seek to end */
    long size = ftell(fp);            /* get position = file size */
    printf("File size: %ld bytes\n", size);

    fseek(fp, 0, SEEK_SET);          /* seek back to start */
    /* SEEK_SET = from start, SEEK_CUR = from current, SEEK_END = from end */

    fclose(fp);
    return 0;
}

Diff: Not Checking fopen Return Value

/* BROKEN: no NULL check on fopen */
#include <stdio.h>

int main(void) {
    FILE *fp = fopen("config.txt", "r");
    /* if file doesn't exist, fp is NULL */

    char line[100];
    fgets(line, sizeof(line), fp);  /* CRASH: dereferences NULL */
    printf("%s", line);

    fclose(fp);
    return 0;
}
/* FIXED: always check fopen result */
#include <stdio.h>

int main(void) {
    FILE *fp = fopen("config.txt", "r");
    if (fp == NULL) {            /* check before any use */
        perror("fopen");         /* prints: "fopen: No such file or directory" */
        return 1;
    }

    char line[100];
    if (fgets(line, sizeof(line), fp) != NULL) {
        printf("%s", line);
    }

    fclose(fp);
    return 0;
}
What was wrong: fopen() returns NULL when it fails (file not found, permission denied, disk error). Passing NULL to fgets dereferences the null pointer — immediate segfault.

Why the fix works: The NULL check catches the failure before any use of fp. perror() prints a human-readable error message combining your label with the system error string (from errno).
⚠ Common Mistakes
  • Forgetting to close files — Each open file consumes a file descriptor. Operating systems have a limit (typically 1024 per process). Not closing files eventually causes fopen to fail with "Too many open files".
  • Using fgetc return value as char — fgetc returns an int, not a char. On a system where char is unsigned, EOF (-1) would wrap to 255, causing an infinite loop. Always declare the return as int.
  • Text mode vs binary mode on Windows — In text mode on Windows, \r\n is translated to \n on read. For binary data, always use "rb"/"wb" to prevent corruption.

Practice Exercises

Basic Line counter

Write a program that counts the number of lines in a text file (passed as a command-line argument argv[1]). Print the count. Handle the case where the file cannot be opened.

Show hint
Use fgets in a while loop, incrementing a counter each iteration. Or count '\n' characters with fgetc. Access the filename via argv[1], check argc >= 2 first.
Intermediate CSV reader

Read a CSV file with columns: name, age, score. Parse each line into a struct and print a formatted table. Handle missing fields gracefully.

Show hint
Use fgets to read each line, then sscanf(line, "%29[^,],%d,%f", name, &age, &score) to parse. The %[^,] reads characters up to (but not including) a comma.
Challenge Binary file editor

Write a simple hex dump tool: read a binary file and print its contents as hex bytes (16 per line) with the ASCII representation on the right (like the xxd command). Print the file offset on the left.

Show hint
Read 16 bytes at a time with fread. Print offset as "%08lx". Print each byte as "%02x ". For the ASCII column: print the byte if it's printable (isprint()), else '.'. Include <ctype.h>.

Checkpoint Quiz

Test your understanding

1. What does fopen return if the file cannot be opened?
fopen returns NULL on failure. Always check for NULL before using the returned FILE*. Use perror() to print the reason.
2. Which mode opens a file for writing and truncates it if it exists?
"w" opens for writing and truncates. "a" appends to the end. "r" is read-only. "x" is not a standard fopen mode.
3. What value does fgetc return at end-of-file?
EOF (defined as -1 in stdio.h) is returned by fgetc at end-of-file or on error. This is why fgetc returns int, not char — to be able to return -1 distinctly.
